[tex]\boxed{a^{b}=\underbrace{a \times a \times a \times \dots \times a}_{sebanyak~b~kali}}[/tex]

[tex]\boxed{a^{-b}=\frac{1}{a^{b}}}[/tex]

[tex]\boxed{a^{\frac{1}{b}}=\sqrt[b]{a}}[/tex]

[tex]\boxed{a^{b}\times a^{c}=a^{(b+c)}}[/tex]

[tex]\boxed{a^{b}\div a^{c}=a^{(b-c)}}[/tex]

[tex]\boxed{(a^{b})^{c}=a^{bc}}[/tex]
